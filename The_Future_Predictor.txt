while True:

        Your_field = input("\nHello, Making your future bright!! In which Field do You want to master? ")
        
        
        if Your_field.strip() and any(char.isalpha() for char in Your_field):
            print(f"\nYour selected Field is {Your_field}")
            break

            print("\nPlease enter a valid field name (text only)")

        continue

            print("\nPlease enter a number between 1 to 100")

        else:
            print(f"\nYou selected top {Skills_percentage}% - good choice!")
